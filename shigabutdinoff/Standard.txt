// Finds whether a variable is an array
func is_array(value)
    return Type(value) = Type([])

// Finds whether a variable is null.
func is_null(value)
    return Type(value) = Type(null)

// Create an array containing a range of elements
func range(start, end)
    var items = []
    for item = start..end
        items.Add(item)
    return items
