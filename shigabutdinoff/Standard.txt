// Finds whether a variable is an array.
func is_array(value)
    return Type(value) = Type([])

// Finds whether a variable is null.
func is_null(value)
    return Type(value) = Type(null)

// Create an array containing a range of elements.
func range(from, to)
    var values = []
    for item = from .. to
        values.Add(item)
    return values

// Return all the values of an array.
func array_values(array)
    var values = []
    for item : array
        values.Add(item)
    return values

// All entries of input equal to false (see converting to boolean) will be removed.
func array_filter(array)
    var values = []
    for item : array
        ? item
            values.Add(item)
    return values

// Checks if the given key or index exists in the array.
func array_key_exists(position, array)
    return position < array.Count()

// Removes duplicate values from an array.
func array_unique(array)
    var values = []
    for item : array
        ? !values.Contains(item)
            values.Add(item)
    return values

// Searches the array for a given value and returns the first corresponding key if successful.
func array_search(needle, haystack)
    position = haystack.IndexOf(needle)
    ? position > 0
        return position
    :
        return false
